import { TranslationKey } from "@myacuvue_thailand_web/services";
import { DatePicker, DatePickerProps, Form } from "antd";
import moment from "moment";
import { FC, useState } from "react";
import { useText } from "../../hooks/useText";
import CalendarIcon, { IconSize } from "../../icons/CalendarIcon";
import ExclamationIcon, {
	ExclamationIconSize,
} from "../../icons/ExclamationIcon";
import Text from "../Text";
import "./index.scss";
import { useService } from "../../hooks/useService";
 
interface IProps {
	name: string;
	value: string;
	onChange: (value: string) => void;
	placeholder: TranslationKey;
	label: TranslationKey;
	errorKey?: TranslationKey;
	alwaysVisibleErrorKey?: TranslationKey;
	className?: string;
	disableBeforeYears?: number;
	format?: string;
	picker: DatePickerProps["picker"];
}
 
const InputError: FC<{ errorKey: TranslationKey }> = ({ errorKey }) => (
	<span>
		<ExclamationIcon size={ExclamationIconSize.SMALL} />{" "}
		<Text textKey={errorKey} />
	</span>
);
 
const GenericDateInput: FC<IProps> = ({
	placeholder,
	label,
	errorKey,
	name,
	value,
	onChange,
	className,
	disableBeforeYears = 0,
	alwaysVisibleErrorKey,
	format,
	picker,
}) => {
	const [isTouched, setIsTouched] = useState(false);
	const displayError = isTouched && errorKey;
 
	const errorToDisplay = alwaysVisibleErrorKey ? (
		<InputError errorKey={alwaysVisibleErrorKey} />
	) : (
		displayError && errorKey && <InputError errorKey={errorKey} />
	);
 
	const yearsAgo = moment().subtract(disableBeforeYears, "year");
 
	const { ClassService } = useService();
	const classNames = ClassService.createClassName(
		"acuvue-date-input",
		className
	);
 
	return (
		<Form.Item
			className={classNames}
			label={<Text textKey={label} />}
			validateStatus={displayError ? "error" : "success"}
			help={errorToDisplay}
		>
			<DatePicker
				format={format}
				picker={picker}
				onBlur={() => setIsTouched(true)}
				name={name}
				value={value ? moment(value, format) : undefined}
				onChange={(momentDate) => {
					onChange(momentDate ? momentDate.format(format) : "");
				}}
				disabledDate={(current) =>
					current && !current.isBefore(yearsAgo)
				}
				className="acuvue-datepicker"
				placeholder={useText(placeholder)}
				suffixIcon={<CalendarIcon size={IconSize.SMALL} />}
			/>
		</Form.Item>
	);
};
 
export default GenericDateInput;
